        matchForm.addEventListener('submit', (e) => {

            e.preventDefault();

            

            const myCharacter = document.getElementById('match-my-character').value;

            const character = document.getElementById('match-character').value;

            const result = document.querySelector('input[name="match-result"]:checked').value;

        function renderStats() {

            const tbody = document.getElementById('stats-table-body');

            const noDataMsg = document.getElementById('no-match-data');

            

            if (matches.length === 0) {

                tbody.innerHTML = '';

                noDataMsg.classList.remove('hidden');

                return;

            }

            

            noDataMsg.classList.add('hidden');


            const stats = {};

            matches.forEach(m => {

                const myChar = m.myCharacter || '未設定';

                const oppChar = m.character;

                const key = `${myChar}_${oppChar}`;


                if (!stats[key]) {

                    stats[key] = { myChar, oppChar, play: 0, win: 0 };

                }

                stats[key].play++;

                if (m.result === 'win') stats[key].win++;

            });


            const statsArray = Object.values(stats).map(data => {

                const winRate = Math.round((data.win / data.play) * 100);

                return { ...data, winRate };

            }).sort((a, b) => b.play - a.play);


            tbody.innerHTML = statsArray.map(stat => `

                <tr class="hover:bg-slate-700 transition-colors">

                    <td class="px-4 py-3 font-medium text-slate-200">${stat.myChar}</td>

                    <td class="px-4 py-3 font-medium text-white">${stat.oppChar}</td>

                    <td class="px-4 py-3 text-right">${stat.play}</td>

                    <td class="px-4 py-3 text-right text-blue-400">${stat.win}</td>

                    <td class="px-4 py-3 text-right font-bold ${stat.winRate >= 50 ? 'text-green-400' : 'text-red-400'}">

                        ${stat.winRate}%

                    </td>

                </tr>

            `).join('');

        }


        function renderMatchHistory() {

            const list = document.getElementById('match-history-list');

            const trendMeter = document.getElementById('trend-meter');

            

            if (matches.length === 0) {

                list.innerHTML = '<div class="text-center text-slate-400 py-4">まだ対戦記録がありません。</div>';

                trendMeter.innerHTML = '';

                return;

            }


            // 最新の10件

            const recentMatches = [...matches].reverse().slice(0, 10);


            // 💡 3. 調子メーターの描画

            const winCount = recentMatches.filter(m => m.result === 'win').length;

            const trendWinRate = Math.round((winCount / recentMatches.length) * 100);

            

            // 丸いドットを生成 (古い試合が左、新しい試合が右になるように並べる)

            const dotsHtml = [...recentMatches].reverse().map(m => {

                return m.result === 'win' 

                    ? '<span class="w-3 h-3 rounded-full bg-blue-500 shadow-[0_0_5px_rgba(59,130,246,0.6)]" title="WIN"></span>'

                    : '<span class="w-3 h-3 rounded-full bg-red-500 shadow-[0_0_5px_rgba(239,68,68,0.6)]" title="LOSE"></span>';

            }).join('');


            trendMeter.innerHTML = `

                <span class="text-xs font-bold text-slate-400 tracking-wider">直近: ${trendWinRate}%</span>

                <div class="flex gap-1 ml-2 bg-slate-900/50 py-1 px-2 rounded-full border border-slate-700">

                    ${dotsHtml}

                </div>

            `;


            // リスト描画

            list.innerHTML = recentMatches.map(m => {

                const d = new Date(m.date);

                const dateStr = `${d.getFullYear()}/${(d.getMonth()+1).toString().padStart(2, '0')}/${d.getDate().toString().padStart(2, '0')}`;

                const isWin = m.result === 'win';

                const resultText = isWin ? 'WIN' : 'LOSE';

                const resultColor = isWin ? 'bg-blue-900/50 text-blue-300 border border-blue-800' : 'bg-red-900/50 text-red-300 border border-red-800';

                const replayBadge = m.replayId ? `<span class="text-xs bg-slate-700 text-slate-300 px-2 py-1 rounded ml-2 font-mono">ID: ${m.replayId}</span>` : '';

                const memoText = m.memo ? `<p class="text-sm text-slate-300 mt-2 whitespace-pre-wrap">${m.memo}</p>` : '';


                return `

                <div class="bg-slate-700/50 p-3 rounded-lg border border-slate-600">

                    <div class="flex justify-between items-center mb-1">

                        <div class="flex items-center flex-wrap gap-y-1">

                            <span class="text-xs font-bold ${resultColor} px-2 py-1 rounded-sm mr-2">${resultText}</span>

                            <span class="text-sm font-medium text-slate-200">${m.myCharacter || '未設定'} <span class="text-slate-500 mx-1">vs</span> ${m.character}</span>

                            ${replayBadge}

                        </div>

                        <span class="text-xs text-slate-400 shrink-0 ml-2">${dateStr}</span>

                    </div>

                    ${memoText}

                </div>

                `;

            }).join('');

        }

        // 簡易トースト通知 (画面上部にフワッと出るやつ)

        function showToast(message) {

            const toast = document.createElement('div');

            toast.className = 'fixed top-4 left-1/2 transform -translate-x-1/2 bg-indigo-600 text-white px-6 py-3 rounded-full shadow-xl font-bold z-50 fade-in';

            toast.textContent = message;

            document.body.appendChild(toast);

            

            setTimeout(() => {

                toast.style.opacity = '0';

                toast.style.transition = 'opacity 0.5s ease';

                setTimeout(() => toast.remove(), 500);

            }, 2000);

        }
